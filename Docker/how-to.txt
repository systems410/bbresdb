
# build the image if it hasnt been built. ("docker image ls" to see docker containers)

docker build -f Docker/Dockerfile -t <imagename> .

# run the docker container 

mkdir ../.bazelcache  # make this to save compiled code 
docker run -d -v .:/app -v ../.bazelcache:/root/.cache/bazel --cap-add SYS_PTRACE --security-opt seccomp=unconfined --name <containername> <imagename> 

# open a shell in the container
docker exec -it <containername> bash
